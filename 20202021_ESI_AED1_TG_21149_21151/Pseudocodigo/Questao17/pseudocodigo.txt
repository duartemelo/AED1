funcao contaCaracter (texto: caracter, caracter: caracter) : inteiro
    Variaveis
        contador: inteiro
    inicio-funcao
        contador <- 0
        enquanto (*texto != '\0')
            se(*texto = caracter)
                contador <- contador + 1
            fim-se
            texto <- texto + 1
        fim-enquanto

        retorna(contador)

    fim-funcao
